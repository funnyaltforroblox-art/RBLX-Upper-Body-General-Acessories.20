-- Visual Highlight Tracking Pointer
local currentHighlightBox = nil

-- Cleans up any existing target highlights safely
local function clearActiveHighlight()
	if currentHighlightBox then
		currentHighlightBox:Destroy()
		currentHighlightBox = nil
	end
end

-- Locates the physical part instance by name inside the current slot's accessory structure
local function findPhysicalPartInCharacter(slotIndex, partName)
	local character = player.Character
	if not character then return nil end
	
	local accessory = character:FindFirstChild("CustomBody_" .. tostring(slotIndex))
	if not accessory then return nil end
	
	if partName == "Handle" then
		return accessory:FindFirstChild("Handle")
	else
		return accessory:FindFirstChild(partName, true)
	end
end

-- Generates or moves a vibrant cyan bounding box onto the chosen active sub-component
local function highlightSelectedSubPart()
	clearActiveHighlight()
	
	local targetPartInstance = findPhysicalPartInCharacter(selectedIndex, selectedSubPartName)
	if targetPartInstance and targetPartInstance:IsA("BasePart") then
		local selectionBox = Instance.new("SelectionBox")
		selectionBox.Name = "Customizer_PartHighlight"
		selectionBox.Color3 = Color3.fromRGB(0, 255, 255) -- Pure Cyan
		selectionBox.LineThickness = 0.04
		selectionBox.Adornee = targetPartInstance
		selectionBox.Parent = targetPartInstance
		
		currentHighlightBox = selectionBox
	end
end

-- Utility function to completely clean up physical objects
local function clearPhysicalAccessories()
	clearActiveHighlight()
	local character = player.Character
	if character then
		for _, child in ipairs(character:GetChildren()) do
			if child:IsA("Accessory") and string.sub(child.Name, 1, 11) == "CustomBody_" then
				child:Destroy()
			end
		end
	end
	table.clear(originalPositions)
	table.clear(originalAnchoredStates)
	isExploded = false
end

-- Render the physical objects onto the character model with Sub-Part transformations
local function applyAllAccessories()
	local character = player.Character
	if not character then return end
	
	local humanoid = character:WaitForChild("Humanoid", 5)
	if not humanoid or humanoid.RigType ~= Enum.HumanoidRigType.R6 then return end
	
	clearPhysicalAccessories()
	
	for index, data in ipairs(accessoryList) do
		local limb = character:FindFirstChild(data.targetPart)
		if limb then
			local success, assetModel = pcall(function()
				return game:GetObjects("rbxassetid://" .. data.id)[1]
			end)
			
			if success and assetModel then
				local handle = assetModel:FindFirstChild("Handle") or assetModel:FindFirstChildWhichIsA("Part", true) or assetModel:FindFirstChildWhichIsA("MeshPart", true)
				if not handle and assetModel:IsA("Model") and assetModel.PrimaryPart then
					handle = assetModel.PrimaryPart
				end
				
				if handle then
					handle.Name = "Handle"
					
					-- Track down and dynamically parse unique component descriptors inside multi-mesh models
					for _, obj in ipairs(assetModel:GetDescendants()) do
						if obj:IsA("BasePart") then
							if not data.partsData[obj.Name] then
								data.partsData[obj.Name] = {offsetX = 0, offsetY = 0, offsetZ = 0, rotateX = 0, rotateY = 0, rotateZ = 0, scaleX = 1, scaleY = 1, scaleZ = 1}
							end
						end
					end

					-- Apply specific multi-dimensional dimension scales directly to geometry layers
					for _, obj in ipairs(assetModel:GetDescendants()) do
						if obj:IsA("BasePart") then
							obj.Anchored = false
							obj.CanCollide = false
							obj.Massless = true
							
							local pData = data.partsData[obj.Name] or {scaleX=1, scaleY=1, scaleZ=1}
							local mesh = obj:FindFirstChildWhichIsA("SpecialMesh")
							if mesh then
								mesh.Scale = Vector3.new(pData.scaleX, pData.scaleY, pData.scaleZ)
							elseif obj:IsA("MeshPart") or not mesh then
								obj.Size = Vector3.new(pData.scaleX, pData.scaleY, pData.scaleZ)
							end
						elseif obj:IsA("Weld") or obj:IsA("ManualWeld") or obj:IsA("WeldConstraint") or obj:IsA("Attachment") then
							obj:Destroy()
						end
					end
					
					local accessory = Instance.new("Accessory")
					accessory.Name = "CustomBody_" .. tostring(index)
					
					if assetModel:IsA("Model") then
						for _, obj in ipairs(assetModel:GetDescendants()) do
							if obj:IsA("BasePart") and obj ~= handle then
								local pData = data.partsData[obj.Name]
								local innerWeld = Instance.new("Weld")
								innerWeld.Name = "ModelInnerWeld"
								innerWeld.Part0 = obj
								innerWeld.Part1 = handle
								
								local posOffset = CFrame.new(pData.offsetX, pData.offsetY, pData.offsetZ)
								local rotOffset = CFrame.Angles(math.rad(pData.rotateX), math.rad(pData.rotateY), math.rad(pData.rotateZ))
								innerWeld.C0 = posOffset * rotOffset
								innerWeld.Parent = obj
							end
						end
						for _, obj in ipairs(assetModel:GetChildren()) do
							if obj:IsA("BasePart") or obj:IsA("Folder") or obj:IsA("Configuration") then
								obj.Parent = accessory
							end
						end
						assetModel:Destroy()
					else
						handle.Parent = accessory
					end
					
					local rootData = data.partsData["Handle"] or {offsetX=0, offsetY=0, offsetZ=0, rotateX=0, rotateY=0, rotateZ=0}
					local weld = Instance.new("Weld")
					weld.Name = "AccessoryWeld"
					weld.Part0 = handle
					weld.Part1 = limb
					
					local positionOffset = CFrame.new(rootData.offsetX, rootData.offsetY, rootData.offsetZ)
					local rotationOffset = CFrame.Angles(math.rad(rootData.rotateX), math.rad(rootData.rotateY), math.rad(rootData.rotateZ))
					
					weld.C0 = positionOffset * rotationOffset
					weld.Parent = handle
					
					accessory.Parent = character
				end
			end
		end
	end
	
	-- Refresh visual highlighter bounds context following component canvas update steps
	task.defer(highlightSelectedSubPart)
end

-- Core Inspection Engine Logic
local function togglePartExplosion()
	local character = player.Character
	if not character then return end

	for _, tween in ipairs(activeTweens) do tween:Cancel() end
	table.clear(activeTweens)

	isExploded = not isExploded

	if isExploded then
		clearActiveHighlight() -- Turn off standard highlighter overlays during explosive mode viewports
		inspectBtn.Text = "COMBINE CUSTOM MODELS"
		inspectBtn.BackgroundColor3 = Color3.fromRGB(150, 35, 35)
		inspectBtnStroke.Color = Color3.fromRGB(255, 100, 100)

		table.clear(originalPositions)
		table.clear(originalAnchoredStates)
		
		for _, child in ipairs(character:GetChildren()) do
			if child:IsA("Accessory") and string.sub(child.Name, 1, 11) == "CustomBody_" then
				local handle = child:FindFirstChild("Handle")
				if handle then
					originalPositions[handle] = handle.CFrame
					originalAnchoredStates[handle] = handle.Anchored
					handle.Anchored = true 

					for _, subPart in ipairs(child:GetDescendants()) do
						if subPart:IsA("BasePart") then
							originalPositions[subPart] = subPart.CFrame
							originalAnchoredStates[subPart] = subPart.Anchored
							subPart.Anchored = true 
						end
					end
				end
			end
		end

		task.wait(0.05) 

		for part, origCFrame in pairs(originalPositions) do
			if part and part.Parent then
				local innerWeld = part:FindFirstChild("ModelInnerWeld") or part:FindFirstChild("AccessoryWeld")
				if innerWeld then innerWeld.Enabled = false end

				local parentAccessory = part:FindFirstAncestorOfClass("Accessory")
				local handle = parentAccessory and parentAccessory:FindFirstChild("Handle")
				local centerPos = handle and handle.Position or character:GetPivot().Position
				
				local direction = (part.Position - centerPos).Unit
				if direction.Magnitude == 0 or tostring(direction) == "NAN, NAN, NAN" then
					direction = Vector3.new(0, 1, 0)
				end

				local explodedPosition = origCFrame.Position + (direction * EXPLODE_DISTANCE)
				local targetCFrame = CFrame.new(explodedPosition) * (origCFrame - origCFrame.Position)

				local tween = TweenService:Create(part, TweenInfo.new(0.4, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {CFrame = targetCFrame})
				table.insert(activeTweens, tween)
				tween:Play()
			end
		end
	else
		inspectBtn.Text = "INSPECT INDIVIDUAL PARTS"
		inspectBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 55)
		inspectBtnStroke.Color = Color3.fromRGB(0, 150, 255)

		for part, origCFrame in pairs(originalPositions) do
			if part and part.Parent then
				local tween = TweenService:Create(part, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {CFrame = origCFrame})
				table.insert(activeTweens, tween)
				
				tween.Completed:Connect(function()
					if part and part.Parent then
						local innerWeld = part:FindFirstChild("ModelInnerWeld") or part:FindFirstChild("AccessoryWeld")
						if innerWeld then innerWeld.Enabled = true end
						task.wait()
						part.Anchored = originalAnchoredStates[part] or false
					end
				end)
				tween:Play()
			end
		end
		task.wait(0.35)
		highlightSelectedSubPart() -- Restore structural tracking colors
	end
end

inspectBtn.MouseButton1Click:Connect(togglePartExplosion)

-- Configuration Vector Redraws
updateEditUI = function()
	local data = accessoryList[selectedIndex]
	if not data then return end
	
	idBox.Text = tostring(data.id)
	
	if not data.partsData[selectedSubPartName] then
		selectedSubPartName = "Handle"
		if not data.partsData["Handle"] then
			data.partsData["Handle"] = {offsetX = 0, offsetY = 0, offsetZ = 0, rotateX = 0, rotateY = 0, rotateZ = 0, scaleX = 1, scaleY = 1, scaleZ = 1}
		end
	end
	
	local pData = data.partsData[selectedSubPartName]
	for prop, box in pairs(inputs) do
		box.Text = tostring(pData[prop] or 0)
	end
	
	for limbName, btn in pairs(targetButtons) do
		if data.targetPart == limbName then
			btn.BackgroundColor3 = Color3.fromRGB(0, 130, 0)
			btn.Text = "[" .. limbName .. "]"
		else
			btn.BackgroundColor3 = Color3.fromRGB(55, 55, 55)
			btn.Text = limbName
		end
	end
	updateSubPartDropdown()
end

local presetNameBox = Instance.new("TextBox")

presetNameBox.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
presetNameBox.TextColor3 = Color3.fromRGB(255, 255, 255)
presetNameBox.Text = "PresetName"
presetNameBox.ClearTextOnFocus = true
presetNameBox.TextSize = 12
presetNameBox.Parent = savePanel

local saveBtn = Instance.new("TextButton")
saveBtn.Size = UDim2.new(0, 60, 1, 0)
saveBtn.Position = UDim2.new(0, 120, 0, 0)
saveBtn.BackgroundColor3 = Color3.fromRGB(110, 80, 20)
saveBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
saveBtn.Text = "SAVE"
saveBtn.Font = Enum.Font.SourceSansBold
saveBtn.TextSize = 13
saveBtn.Parent = savePanel

saveBtn.MouseButton1Click:Connect(function()
	local name = presetNameBox.Text
	if name ~= "" and HAS_FILESYSTEM then
		local json = HttpService:JSONEncode(accessoryList)
		local success = pcall(function()
			writefile(FOLDER_NAME .. "/" .. name .. ".txt", json)
		end)
		if success then
			presetNameBox.Text = "Saved!"
			task.wait(1)
			presetNameBox.Text = "PresetName"
			updatePresetUI()
		end
	else
		presetNameBox.Text = "No File Auth"
		task.wait(1)
		presetNameBox.Text = "PresetName"
	end
end)

updatePresetUI = function()
	presetPanel:ClearAllChildren()
	if not HAS_FILESYSTEM then return end
	
	local pLayout = Instance.new("UIListLayout")
	pLayout.Padding = UDim.new(0, 4)
	pLayout.Parent = presetPanel
	
	local success, files = pcall(function() return listfiles(FOLDER_NAME) end)
	if not success or not files then return end
	
	for _, path in ipairs(files) do
		local filename = string.match(path, "([^/]+)%.txt$") or string.match(path, "([^\\]+)%.txt$")
		if filename then
			local pFrame = Instance.new("Frame")
			pFrame.Size = UDim2.new(1, 0, 0, 25)
			pFrame.BackgroundTransparency = 1
			pFrame.Parent = presetPanel
			
			local loadBtn = Instance.new("TextButton")
			loadBtn.Size = UDim2.new(1, 0, 1, 0)
			loadBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
			loadBtn.TextColor3 = Color3.fromRGB(230, 230, 230)
			loadBtn.Text = filename
			loadBtn.TextSize = 13
			loadBtn.Parent = pFrame
			
			loadBtn.MouseButton1Click:Connect(function()
				local dataSuccess, content = pcall(function() return readfile(path) end)
				if dataSuccess then
					local tableSuccess, decoded = pcall(function() return HttpService:JSONDecode(content) end)
					if tableSuccess and type(decoded) == "table" then
						accessoryList = decoded
						selectedIndex = 1
						selectedSubPartName = "Handle"
						updateSlotUI()
						updateEditUI()
						applyAllAccessories()
					end
				end
			end)
		end
	end
end

updateSlotUI()
